Test_SNF_ECMS_16_FormGenerator_ecmdemo
open /ecmdemo/acme/
windowMaximize
echo * Log in
waitForElementPresent link=Login
click link=Login
waitForElementPresent username
type username root
type password gtn
waitForElementPresent link=Sign in
clickAndWait link=Sign in
echo * Create a File in  Sites management/ acme
echo -- Go to Content Explorer
waitForElementPresent link=Content Explorer
clickAndWait link=Content Explorer
echo -- Click Show Drives
waitForElementPresent css=a.NoneDrivesButton
click css=a.NoneDrivesButton
echo -- Choose link=Sites Management
waitForElementPresent link=Sites Management
click link=Sites Management
echo -- Choose acme
waitForElementPresent css=a[title="acme "]
click css=a[title="acme "]
echo -- Click Add Content
waitForElementPresent link=New Content
click link=New Content
echo -- Choose File template
waitForElementPresent //div[@title='File']
click //div[@title='File']
waitForElementPresent id=name Test_SNF_ECMS_16
type id=name Test_SNF_ECMS_16
waitForElementPresent //td[@id="cke_contents_contentHtml"]/iframe
selectFrame //td[@id="cke_contents_contentHtml"]/iframe
type //html/body File created in Test_SNF_ECMS_16
pause 3000
selectFrame relative=top
waitForElementPresent link=Save & Close
click link=Save & Close
pause 3000
echo * Edit File
waitForElementPresent xpath=(//a[contains(text(),'Edit')])[5]
click xpath=(//a[contains(text(),'Edit')])[5]
echo -- Edit Content to "Edit: Test_SNF_ECMS_16"
waitForElementPresent //td[@id="cke_contents_contentHtml"]/iframe
selectFrame //td[@id="cke_contents_contentHtml"]/iframe
type //html/body Edit: Test_SNF_ECMS_16
pause 3000
selectFrame relative=top
waitForElementPresent link=Save & Close
click link=Save & Close
pause 3000
echo * Delete File
waitForElementPresent link=Test_SNF_ECMS_16
mouseDownRight link=Test_SNF_ECMS_16
waitForElementPresent link=Delete
click link=Delete
waitForElementPresent link=OK
click link=OK
echo -- Verify File is deleted
waitForTextNotPresent link=Test_SNF_ECMS_16
verifyTextNotPresent link=Test_SNF_ECMS_16
pause 3000
echo * Log out
waitForElementPresent link=Sign out
clickAndWait link=Sign out
