waitForElementPresent //td/label[@for='userName']/following::td/input[@value='john']
verifyElementPresent //td/label[@for='userName']/following::td/input[@value='john']
verifyElementPresent //td/label[@for='firstName']/following::td/input[@value='John']

waitForElementPresent //td/label[@for='userName']/following::td/input[@value='john']
verifyElementPresent //td/label[@for='userName']/following::td/input[@value='john']
verifyElementPresent //td/label[@for='firstName']/following::td/input[@value='John']
