Test_POR_11_002_CheckWhenChangeToPortalThatIsNotPublic
setSpeed 1000
open /portal/intranet
echo -- Check when change to a portal that is not public--
windowMaximize
click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username john
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
pause 10000
waitForElementPresent link=Sites
clickAndWait link=Sites
waitForElementPresent link=Add New Portal
click link=Add New Portal
waitForElementPresent name
type name Test_POR_11_002
echo -- Choose properties--
waitForElementPresent //div[text()='Properties']
click //div[text()='Properties']
echo -- Choose Portal Template --
waitForElementPresent //div[text()='Portal Templates']
click //div[text()='Portal Templates']
waitForElementPresent //div[text()='Intranet Portal']
click //div[text()='Intranet Portal']
echo -- Choose Permision Setting --
waitForElementPresent //div[text()='Permission Settings']
click //div[text()='Permission Settings']
echo -- Access Permision Setting--
waitForElementPresent link=Add Permission
click link=Add Permission
waitForElementPresent link=Platform
click link=Platform
waitForElementPresent //div[@id='ListPermissionSelector']//a[@title="Administration"]
click //div[@id='ListPermissionSelector']//a[@title="Administration"]
waitForElementPresent link=manager
click link=manager
echo -- Edit Permision Setting --
waitForElementPresent link=Edit Permission Settings
click link=Edit Permission Settings
waitForElementPresent link=Select Permission
click link=Select Permission
waitForElementPresent link=Platform
click link=Platform
waitForElementPresent //div[@id='PermissionSelector']//a[@title="Administration"]
click //div[@id='PermissionSelector']//a[@title="Administration"]
waitForElementPresent //div[@id='PermissionSelector']//a[@title='validator']
click //div[@id='PermissionSelector']//a[@title='validator']
waitForElementPresent //a[contains(text(),'Save')]
clickAt //a[contains(text(),'Save')] 2,2
pause 20000
refreshAndWait
waitForElementPresent link=Test_POR_11_002
clickAndWait link=Test_POR_11_002
echo -- Check displaying after sign out when using un-public portal --
waitForElementPresent link=Logout
clickAndWait link=Logout
assertLocation http://localhost:8080/portal/dologin?initialURI=/portal/Test_POR_11_002/
echo -- Check the next login after choosing un-public portal--
open http://localhost:8080/portal/private/Test_POR_11_002/
waitForElementPresent username
type username john
type password gtn
click link=Sign in
pause 5000
assertLocation http://localhost:8080/portal/Test_POR_11_002/
clickAndWait link=intranet
waitForElementPresent link=Sites
click link=Sites
waitForElementPresent //td/div[text()='Test_POR_11_002']/following::td/a[text()='Delete']
clickAt //td/div[text()='Test_POR_11_002']/following::td/a[text()='Delete'] 2,2
waitForConfirmationPresent
assertConfirmation Are you sure to delete this portal?
pause 60000
waitForTextNotPresent Test_POR_11_002
verifyTextNotPresent Test_POR_11_002
waitForElementPresent link=Logout
clickAndWait link=Logout
