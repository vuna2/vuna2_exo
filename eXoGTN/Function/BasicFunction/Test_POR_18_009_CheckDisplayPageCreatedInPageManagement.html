setSpeed 1000
open /portal/intranet
windowMaximize
click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username root
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
pause 10000
echo -- Add new page for group--
waitForElementPresent link=Pages
clickAndWait link=Pages
waitForElementPresent //div[@id='UIPageBrowser']//a[text()="Add New Page"]
click //div[@id='UIPageBrowser']//a[text()="Add New Page"]
waitForElementPresent ownerType
select ownerType label=group
waitForElementPresent ownerId
select ownerId label=/organization/management/executive-board
waitForElementPresent name
type name TEST_POR_18_009
type title TEST_POR_18_009
echo -- Choose Page Layout --
waitForElementPresent //div[@id='UIMaskWorkspace']//div[text()="Page Layout"]
click //div[@id='UIMaskWorkspace']//div[text()="Page Layout"]
echo -- Choose Permission Setting --
waitForElementPresent //div[@id='UIMaskWorkspace']//div[text()="Permission Settings"]
click //div[@id='UIMaskWorkspace']//div[text()="Permission Settings"]
echo -- Access Permission Setting --
waitForTextPresent Access Permission Setting
verifyTextPresent Access Permission Setting
echo -- Edit Permission Setting --
waitForElementPresent link=Edit Permission Settings
click link=Edit Permission Settings
pause 5000
verifyText css=div.PermissionInfoTitle Current Permission
waitForElementPresent link=Save
click link=Save
waitForElementPresent link=Logout
clickAndWait link=Logout

type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']

waitForElementPresent link=Logout
clickAndWait link=Logout

echo -- Search by Title

click //a[@title='Quick Search']
