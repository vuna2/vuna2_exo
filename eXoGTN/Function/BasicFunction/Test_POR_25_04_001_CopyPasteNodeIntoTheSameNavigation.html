Test_POR_25_04_001_CopyPasteNodeIntoTheSameNavigation
setSpeed 1000
open /portal/intranet
echo --Login portal--
windowMaximize
click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username john
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
pause 10000
echo -- Go to Group navigation --
clickAndWait link=Group Sites
waitForElementPresent //td/div[text()='Administration']/ancestor::tr/td/a[text()='Edit Navigation']
click //td/div[text()='Administration']/ancestor::tr/td/a[text()='Edit Navigation']
echo --Copy a node--
waitForElementPresent //a[@title='Add Users']
mouseDownRight xpath=//div[@id='UINavigationNodeSelector']//a[@class='NodeIcon DefaultPageIcon' and @title='Add Users']
waitForElementPresent link=Copy Node
click link=Copy Node
echo --Paste node into another node in the same navigation--
waitForElementPresent //div[@id='UINavigationNodeSelector']//a[@title='Monitoring']
click //div[@id='UINavigationNodeSelector']//a[@title='Monitoring']
waitForElementPresent //div[@id='UINavigationNodeSelector']//a[@title='Monitoring']
mouseDownRight //div[@id='UINavigationNodeSelector']//a[@title='Monitoring']
waitForElementPresent xpath=(//a[contains(text(),'Paste Node')])[2]
click xpath=(//a[contains(text(),'Paste Node')])[2]
echo --Delete node--
waitForElementPresent //div[@id='UINavigationNodeSelector']//a[@title='Add Users']
click //div[@id='UINavigationNodeSelector']//a[@title='Add Users']
mouseDownRight //div[@id='UINavigationNodeSelector']//a[@title='Add Users']
waitForElementPresent link=Delete Node
click link=Delete Node
assertConfirmation Are you sure to delete this node?
waitForElementPresent link=Save
click link=Save
echo --Sign out --
waitForElementPresent link=Logout
clickAndWait link=Logout
