Test_SNF_PRL_01_SignInSignOut
open /portal/intranet
echo -SignIn/SignOut-
windowMaximize
echo -Login by Administrator-
click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username root
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
echo -- Go to My Profile --
waitForElementPresent link=My Profile
clickAndWait link=My Profile
echo -- Verify user name --
waitForText //td/label[@for='username']/following::td[contains(text(),'root')] root
verifyText //td/label[@for='username']/following::td[contains(text(),'root')] root
echo -- Verify First Name --
waitForText //td/label[@for='firstName']/following::td[contains(text(),'Root')] Root
verifyText //td/label[@for='firstName']/following::td[contains(text(),'Root')] Root
echo -- Verify Last Name --
waitForText //td/label[@for='lastName']/following::td[contains(text(),'Root')] Root
verifyText //td/label[@for='lastName']/following::td[contains(text(),'Root')] Root
echo -- Verify Email --
waitForText //td/label[@for='email']/following::td/a[contains(text(),'root@localhost')] root@localhost
verifyText //td/label[@for='email']/following::td/a[contains(text(),'root@localhost')] root@localhost
echo -- Logout --
waitForElementPresent link=Logout
clickAndWait link=Logout
