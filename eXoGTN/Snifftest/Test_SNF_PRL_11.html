echo -- Memberships Management--

click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username root
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
echo -- Go to Users and group management --
clickAndWait link=Group and Roles
echo -- Choose Membership Management --
waitForElementPresent link=Membership Management
click link=Membership Management
waitForElementPresent name
type name Test_SNF_PRL_11
type description Test_SNF_PRL_11
waitForElementPresent link=Save
click link=Save
waitForTextPresent Test_SNF_PRL_11
verifyTextPresent Test_SNF_PRL_11

waitForElementPresent link=Group Management
click link=Group Management
waitForElementPresent link=Organization
click link=Organization

click //div[@id='UIOrganizationPortlet']//a[@title='Management']
waitForElementPresent link=Executive Board
click link=Executive Board
waitForElementPresent id=username
type id=username john
waitForElementPresent name=membership
select name=membership label=Test_SNF_PRL_11
waitForElementPresent link=Save
click link=Save
waitForElementPresent //div[text()='john']/following::div[text()='Test_SNF_PRL_11']

click link=Add Permission

click link=Platform

click //div[@id='ListPermissionSelector']//a[@title='Administration']

click link=Test_SNF_PRL_11

click link=Edit Permission Settings
