Test_SNF_PRL_52_ManageTab
open /portal/intranet
echo -- Manage Tab--
windowMaximize
click //b[contains(text(),'Sign in')]
waitForElementPresent username
type username john
type password gtn
clickAndWait //a[@id='UIPortalLoginFormAction']
waitForElementPresent link=Dashboard
clickAndWait link=Dashboard
echo -- Add new tab--
waitForElementPresent css=a.AddDashboard
click css=a.AddDashboard
type //div[@id='UITabPaneDashboard']/div[3]/input Test_SNF_PRL_52
keyPress //div[@id='UITabPaneDashboard']/div[3]/input \13
pause 5000
waitForTextPresent Test_SNF_PRL_52
verifyTextPresent Test_SNF_PRL_52
echo -- Edit tab--
doubleClick //div[@id='UITabPaneDashboard']/div[3]/span
type //div[@id='UITabPaneDashboard']/div[3]/input Test_SNF_PRL_52_edit
keyPress //div[@id='UITabPaneDashboard']/div[3]/input \13
waitForTextPresent Test_SNF_PRL_52_edit
verifyTextPresent Test_SNF_PRL_52_edit
echo -- Delete tab--
clickAndWait css=div.UITab.SelectedTab > a.CloseIcon
assertConfirmation Are you sure to remove this dashboard?
waitForTextNotPresent Test_SNF_PRL_52_edit
verifyTextNotPresent Test_SNF_PRL_52_edit
pause 30000
echo --Add gadgets into dashboard page
clickAndWait link=Dashboard
waitForElementPresent link=Add Gadgets
click link=Add Gadgets
echo --Input URL --
waitForElementPresent //input[@id='url']
type //input[@id='url'] http://www.labpixies.com/campaigns/dates/dates.xml
waitForElementPresent //img[@title='Add Gadget']
click //img[@title='Add Gadget']
echo -- Close Dashboard Workspace--
waitForElementPresent css=a.CloseButton
click css=a.CloseButton
pause 3000
echo -- Delete gadget --
click css=span.MaximizeGadget.IconControl
waitForElementPresent css=span.CloseGadget.IconControl
click css=span.CloseGadget.IconControl
assertConfirmation Are you sure to delete this gadget?
pause 1000
echo -- Drag & drop gadget --
waitForElementPresent link=Add Gadgets
click link=Add Gadgets
waitForElementPresent xpath=//div[@class='GadgetTitle' and @title='Calculator']
dragAndDropToObject xpath=//div[@class='GadgetTitle' and @title='Calculator'] xpath=//div[@class='UIDashboardContainer']
echo -- Change position of gadget --
waitForElementPresent xpath=//div[@class='UIDashboardContainer']/div/div/div[3]
waitForElementPresent //span[text()='Calculator']
dragAndDropToObject //span[text()='Calculator'] //div[text()='Drag your gadgets here.']
echo -- Close Dashboard Workspace--
waitForElementPresent css=a.CloseButton
click css=a.CloseButton
pause 30000
echo --Delete gadget--
waitForElementPresent css=span.CloseGadget.IconControl
click css=span.CloseGadget.IconControl
assertConfirmation Are you sure to delete this gadget?
waitForElementPresent link=Logout
clickAndWait link=Logout
